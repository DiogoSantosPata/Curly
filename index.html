C.U.R.L.Y.

int cx = 300; 
int cy = 300;
int angle = 0;
int mag = 100;

void setup()
{
  size(600, 600);  
}


void draw()
{
  rect(cx, cy, sin(radians(angle))*mag, cos(radians(angle))*mag );
  delay(20);
  
  angle+=4;
  mag+=random(-10,10);
}

  

  	
float n_x,n_y,xx,yy = 300;

void setup()
{
  size(600,600);
}

void draw()
{
  n_x = random(100,500);
  n_y = random(100,500);  
  line(  xx,yy,n_x,n_y );
  
  xx = n_x;
  yy = n_y;
  
  rect(0, 0, 600, 600);
  fill(255, 20);
}
