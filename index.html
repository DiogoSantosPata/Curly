int cx = 300; 
int cy = 300;
int angle = 0;
int mag = 100;
float n_ang;

void setup()
{
  size(600, 600);  
}

void draw()
{
  n_ang = radians(angle);
  rect(cx, cy, sin(n_ang)*mag, cos(n_ang)*mag );
  delay(20);
  
  angle+=4;
  mag+=random(-10,10);
}

float cx, cy;

{
  angle+=5;
  background(255);
  cx = 300 + sin(radians(angle))*50; 
  cy = 300 +  cos(radians(angle))*50;
  line(cx, 0, cx, 600);
  line(cx+10, 0, cx+10, 600);
  line(cx-10, 0, cx-10, 600);
  line(100, cy-100, 500, cy-100);
  line(100, cy+100, 500, cy+100);
  fill(255,0,0);
